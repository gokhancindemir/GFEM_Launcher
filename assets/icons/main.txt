const { app, BrowserWindow, Tray, Menu, ipcMain } = require('electron');
const path = require('path');
const { exec } = require('child_process'); // Node.js'in kendi çekirdek komut aracı (Paket gerektirmez!)

let win;
let tray = null;
let isQuitting = false;

function createWindow () {
    win = new BrowserWindow({
        width: 1200,
        height: 800,
        webPreferences: {
            nodeIntegration: true,
            contextIsolation: false
        }
    });

    win.setMenuBarVisibility(false);
    win.loadFile('index.html');

    win.on('close', function (event) {
        if (!isQuitting) {
            event.preventDefault();
            win.hide();
        }
        return false;
    });
}

app.whenReady().then(() => {
    createWindow();

    // --- DIŞ PAKETSİZ, DOĞRUDAN POWERSHELL İLE DİSK OKUMA ---
    ipcMain.handle('get-disks', () => {
        return new Promise((resolve) => {
            // Windows'a sadece yerel diskleri (DriveType=3) bulmasını söyleyen PowerShell komutu
            const psCommand = `powershell -NoProfile -Command "Get-CimInstance Win32_LogicalDisk | Where-Object DriveType -eq 3 | Select-Object DeviceID, FreeSpace, Size | ConvertTo-Json"`;
            
            exec(psCommand, (error, stdout, stderr) => {
                if (error) {
                    console.error("PowerShell Hatası:", error);
                    resolve(null);
                    return;
                }
                try {
                    let diskData = JSON.parse(stdout);
                    // Eğer bilgisayarda tek bir disk (sadece C) varsa, onu da liste formatına çevir
                    if (!Array.isArray(diskData)) {
                        diskData = [diskData];
                    }
                    resolve(diskData);
                } catch (e) {
                    console.error("JSON çevirme hatası:", e);
                    resolve(null);
                }
            });
        });
    });

    // --- SİSTEM TEPSİSİ (TRAY) AYARLARI ---
    const iconPath = path.join(__dirname, 'assets', 'icons', 'tray-icon.png');
    tray = new Tray(iconPath);
    
    const contextMenu = Menu.buildFromTemplate([
        { label: 'Göster / Aç', click: function() { win.show(); } },
        { label: 'Tamamen Kapat', click: function() { 
            isQuitting = true; 
            app.quit(); 
        }}
    ]);
    
    tray.setToolTip('Launcher_GFEM');
    tray.setContextMenu(contextMenu);

    tray.on('double-click', () => {
        win.show();
    });
});

ipcMain.on('quit-app', () => {
    isQuitting = true;
    app.quit();
});
