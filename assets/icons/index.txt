    <script>
        // Electron kütüphanesini en başta tanımlıyoruz
        const { ipcRenderer } = require('electron');

        // --- 1. MENÜ VE BUTON GEÇİŞLERİ ---
        const menuItems = document.querySelectorAll('.nav-menu li');
        const sections = document.querySelectorAll('.content-section');
        const actionButtons = document.querySelectorAll('.action-buttons .btn');

        menuItems.forEach(item => {
            item.addEventListener('click', () => {
                menuItems.forEach(i => i.classList.remove('active'));
                item.classList.add('active');

                sections.forEach(sec => sec.style.display = 'none');
                const targetId = item.getAttribute('data-target');
                document.getElementById(targetId).style.display = 'block';

                const targetBtnId = item.getAttribute('data-btn');
                
                actionButtons.forEach(btn => {
                    if (targetBtnId === 'all') {
                        btn.style.display = 'flex'; 
                    } else if (btn.id === targetBtnId) {
                        btn.style.display = 'flex'; 
                    } else {
                        btn.style.display = 'none'; 
                    }
                });
            });
        });

        // --- 2. DİSK SÜRÜCÜLERİNİ OKUMA VE ÇİZME (NATIVE POWERSHELL SİSTEMİ) ---
        async function renderDrives() {
            const driveContainer = document.getElementById('dynamic-drives');
            
            try {
                const disks = await ipcRenderer.invoke('get-disks');

                if (!disks) {
                    throw new Error("Motor disk bilgisini okuyamadı.");
                }

                driveContainer.innerHTML = ''; 

                disks.forEach(disk => {
                    // Biçimlendirilmemiş boş diskleri atla
                    if (!disk.Size) return; 

                    const letter = disk.DeviceID; // 'C:' veya 'D:'
                    
                    // PowerShell bayt cinsinden verir, biz GB'a çeviriyoruz
                    const totalGB = (disk.Size / 1073741824).toFixed(1);
                    const freeGB = (disk.FreeSpace / 1073741824).toFixed(1);
                    
                    // Doluluk yüzdesini hesaplama
                    const usedBytes = disk.Size - disk.FreeSpace;
                    const percent = ((usedBytes / disk.Size) * 100).toFixed(0) + '%'; 
                    
                    const isC = letter.toUpperCase().includes('C:');
                    const iconName = isC ? 'drive_c.png' : 'drive_d.png';
                    const typeName = isC ? 'System Drive' : 'Local Drive';
                    const barColor = isC ? '#5C5CFF' : '#1E8E3E';

                    const cardHTML = `
                        <div class="drive-card">
                            <div class="drive-header">
                                <div class="drive-icon">
                                    <img src="assets/icons/${iconName}" alt="Drive ${letter}">
                                </div>
                                <div class="drive-title-area">
                                    <span class="drive-title">Local Disk (${letter})</span>
                                    <span class="drive-type">${typeName}</span>
                                </div>
                            </div>
                            <div class="drive-progress">
                                <div class="progress-track">
                                    <div class="progress-fill" style="width: ${percent}; background-color: ${barColor};"></div>
                                </div>
                            </div>
                            <div class="drive-details">
                                <span>${freeGB} GB free of ${totalGB} GB</span>
                            </div>
                        </div>
                    `;
                    
                    driveContainer.innerHTML += cardHTML;
                });
                
            } catch (error) {
                console.error("Arayüz hatası:", error);
                driveContainer.innerHTML = '<p style="color:#E53935; font-size:14px; margin-top:10px;">Cannot load disk information.</p>';
            }
        }

        // --- 3. PROGRAMI KAPATMA (EXIT) İŞLEMİ ---
        const exitButton = document.getElementById('btn-exit');
        if (exitButton) {
            exitButton.addEventListener('click', () => {
                ipcRenderer.send('quit-app');
            });
        }

    </script>
